# ATBS 前端架构说明

## 1. Vue3 Composition API 选型理由

### 1.1 Composition API vs Options API 对比

| 对比维度 | Composition API | Options API | 选型理由 |
|:--------|:----------------|:------------|:---------|
| **代码组织** | 按功能逻辑组织 | 按选项类型组织 | Composition更直观 |
| **逻辑复用** | Composables函数 | Mixins | Composition无命名冲突 |
| **TypeScript支持** | 优秀 | 一般 | Composition类型推断更好 |
| **代码压缩** | 更好 | 一般 | Composition tree-shaking友好 |
| **学习曲线** | 较陡 | 平缓 | 但长期收益更高 |
| **生态支持** | 未来主流 | 传统方式 | Vue3官方推荐 |

### 1.2 选择 Composition API 的核心原因

### 2.2 Composables实现示例

```typescript
// src/composables/usePrediction.ts
import { ref, computed } from 'vue'
import { predictionApi } from '@/api/prediction'
import type { 
  PredictionRequest, 
  PredictionResponse, 
  ModelInfo,
  TrainingTask 
} from '@/types/prediction'

/**
 * 预测服务组合式函数
 * 封装预测相关的所有逻辑
 */
export function usePrediction() {
  // ========== State ==========
  const forecastData = ref<PredictionResponse | null>(null)
  const modelInfo = ref<ModelInfo | null>(null)
  const currentTask = ref<TrainingTask | null>(null)
  const loading = ref(false)
  const error = ref<string | null>(null)
  
  // ========== Getters ==========
  const hasForecast = computed(() => !!forecastData.value)
  const hasModel = computed(() => !!modelInfo.value)
  const isTaskRunning = computed(() => 
    currentTask.value?.status === 'processing'
  )
  const taskProgress = computed(() => 
    currentTask.value?.progress || 0
  )
  
  // ========== Actions ==========
  
  /**
   * 获取客流预测
   */
  const getForecast = async (params: PredictionRequest) => {
    loading.value = true
    error.value = null
    
    try {
      const data = await predictionApi.getForecast(params)
      forecastData.value = data
      return data
    } catch (err) {
      error.value = err instanceof Error ? err.message : '预测失败'
      throw err
    } finally {
      loading.value = false
    }
  }
  
  /**
   * 获取模型信息
   */
  const fetchModelInfo = async () => {
    loading.value = true
    
    try {
      const data = await predictionApi.getModelInfo()
      modelInfo.value = data
      return data
    } finally {
      loading.value = false
    }
  }
  
  /**
   * 创建训练任务
   */
  const createTrainingTask = async () => {
    loading.value = true
    
    try {
      const task = await predictionApi.createTrainingTask()
      currentTask.value = task
      
      // 开始轮询任务状态
      startPollingTaskStatus(task.taskId)
      
      return task
    } finally {
      loading.value = false
    }
  }
  
  /**
   * 轮询任务状态
   */
  let pollingTimer: number | null = null
  
  const startPollingTaskStatus = (taskId: string) => {
    // 清除之前的定时器
    stopPollingTaskStatus()
    
    const poll = async () => {
      try {
        const task = await predictionApi.getTaskStatus(taskId)
        currentTask.value = task
        
        // 任务完成或失败时停止轮询
        if (task.status === 'completed' || task.status === 'failed') {
          stopPollingTaskStatus()
          
          // 任务完成时刷新模型信息
          if (task.status === 'completed') {
            await fetchModelInfo()
          }
        } else {
          // 继续轮询
          pollingTimer = window.setTimeout(poll, 2000)
        }
      } catch (err) {
        console.error('轮询任务状态失败:', err)
        stopPollingTaskStatus()
      }
    }
    
    poll()
  }
  
  const stopPollingTaskStatus = () => {
    if (pollingTimer) {
      clearTimeout(pollingTimer)
      pollingTimer = null
    }
  }
  
  /**
   * 清除预测数据
   */
  const clearForecast = () => {
    forecastData.value = null
    error.value = null
  }
  
  return {
    // State
    forecastData,
    modelInfo,
    currentTask,
    loading,
    error,
    // Getters
    hasForecast,
    hasModel,
    isTaskRunning,
    taskProgress,
    // Actions
    getForecast,
    fetchModelInfo,
    createTrainingTask,
    clearForecast
  }
}
```

```typescript
// src/stores/modules/auth.ts
import { defineStore } from 'pinia'
import { ref, computed } from 'vue'
import { useRouter } from 'vue-router'
import { authApi } from '@/api/auth'
import type { UserInfo, LoginForm } from '@/types/auth'

export const useAuthStore = defineStore('auth', () => {
  const router = useRouter()
  
  // ========== State ==========
  const token = ref<string>('')
  const userInfo = ref<UserInfo | null>(null)
  const permissions = ref<string[]>([])
  
  // ========== Getters ==========
  const isLoggedIn = computed(() => !!token.value)
  const username = computed(() => userInfo.value?.username || '')
  const avatar = computed(() => userInfo.value?.avatar || '')
  const hasPermission = computed(() => (permission: string) => {
    return permissions.value.includes(permission)
  })
  
  // ========== Actions ==========
  
  /**
   * 登录
   */
  async function login(form: LoginForm) {
    const res = await authApi.login(form)
    token.value = res.token
    userInfo.value = res.userInfo
    permissions.value = res.permissions || []
    return res
  }
  
  /**
   * 登出
   */
  async function logout() {
    try {
      await authApi.logout()
    } finally {
      // 清除状态
      token.value = ''
      userInfo.value = null
      permissions.value = []
      
      // 跳转到登录页
      router.push('/login')
    }
  }
  
  /**
   * 获取用户信息
   */
  async function fetchUserInfo() {
    if (!token.value) return
    
    try {
      const res = await authApi.getUserInfo()
      userInfo.value = res
      permissions.value = res.permissions || []
      return res
    } catch (error) {
      // Token无效，重新登录
      logout()
      throw error
    }
  }
  
  return {
    token,
    userInfo,
    permissions,
    isLoggedIn,
    username,
    avatar,
    hasPermission,
    login,
    logout,
    fetchUserInfo
  }
})
```

## 6. 总结

ATBS前端架构采用Vue 3 Composition API，具有以下优势：

1. **逻辑复用**：通过Composables实现业务逻辑的高度复用
2. **类型安全**：TypeScript全栈类型约束，减少运行时错误
3. **代码组织**：按功能聚合代码，提高可维护性
4. **性能优化**：响应式系统 + 懒加载 + 虚拟列表
5. **状态管理**：Pinia提供简洁高效的全局状态管理

这种架构设计使得代码更加模块化、可测试、可维护，适合中大型前端项目的长期发展。
