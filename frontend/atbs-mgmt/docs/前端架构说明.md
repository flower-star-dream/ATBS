# ATBS 管理端前端架构说明

## 1. 架构概述

### 1.1 架构设计原则

ATBS管理端前端采用**分层架构设计**，遵循以下原则：

1. **单一职责**：每个模块只负责一个明确的功能
2. **高内聚低耦合**：模块内部紧密相关，模块间依赖最小化
3. **可测试性**：各层独立，便于单元测试
4. **可维护性**：清晰的代码结构，易于理解和维护

### 3.1 状态管理 (Pinia)

#### 3.1.1 Store设计

```typescript
// stores/app.ts - 应用状态
export const useAppStore = defineStore('app', () => {
  // State
  const sidebarCollapsed = ref(false)
  const menuList = ref<MenuItem[]>([...])
  
  // Getters
  const activeMenu = computed(() => {
    return menuList.value.find(item => item.path === route.path)
  })
  
  // Actions
  const toggleSidebar = () => {
    sidebarCollapsed.value = !sidebarCollapsed.value
  }
  
  return {
    sidebarCollapsed,
    menuList,
    activeMenu,
    toggleSidebar
  }
})

// stores/employee.ts - 员工状态
export const useEmployeeStore = defineStore('employee', () => {
  // State
  const employeeInfo = ref<EmployeeInfo | null>(null)
  const permissions = ref<string[]>([])
  
  // Getters
  const isLoggedIn = computed(() => !!employeeInfo.value)
  const hasPermission = (permission: string) => {
    return permissions.value.includes(permission)
  }
  
  // Actions
  const setEmployeeInfo = (info: EmployeeInfo) => {
    employeeInfo.value = info
  }
  
  const logout = () => {
    employeeInfo.value = null
    localStorage.removeItem('access_token')
  }
  
  return {
    employeeInfo,
    permissions,
    isLoggedIn,
    hasPermission,
    setEmployeeInfo,
    logout
  }
}, {
  persist: true  // 持久化到localStorage
})
```

### 3.2 组合式函数 (Composables)

```typescript
// composables/useTable.ts - 表格逻辑封装
export function useTable<T>() {
  // 数据
  const tableData = ref<T[]>([])
  const loading = ref(false)
  
  // 分页
  const pagination = reactive({
    page: 1,
    pageSize: 10,
    total: 0
  })
  
  // 搜索
  const searchForm = reactive({
    keyword: ''
  })
  
  // 获取数据
  const fetchData = async (api: () => Promise<any>) => {
    loading.value = true
    try {
      const res = await api()
      tableData.value = res.list
      pagination.total = res.total
    } finally {
      loading.value = false
    }
  }
  
  // 分页变化
  const handlePageChange = (page: number) => {
    pagination.page = page
    fetchData()
  }
  
  // 搜索
  const handleSearch = () => {
    pagination.page = 1
    fetchData()
  }
  
  return {
    tableData,
    loading,
    pagination,
    searchForm,
    fetchData,
    handlePageChange,
    handleSearch
  }
}

// composables/useForm.ts - 表单逻辑封装
export function useForm<T>(options: {
  initialValues: T
  validate?: (values: T) => boolean
  submit: (values: T) => Promise<void>
}) {
  const formData = reactive({ ...options.initialValues })
  const loading = ref(false)
  const errors = ref<Record<string, string>>({})
  
  const handleSubmit = async () => {
    // 验证
    if (options.validate && !options.validate(formData as T)) {
      return
    }
    
    loading.value = true
    try {
      await options.submit(formData as T)
    } finally {
      loading.value = false
    }
  }
  
  const resetForm = () => {
    Object.assign(formData, options.initialValues)
    errors.value = {}
  }
  
  return {
    formData,
    loading,
    errors,
    handleSubmit,
    resetForm
  }
}
```

### 4.1 API封装

```typescript
// utils/request.ts - 请求封装

// 创建请求实例
const createRequestInstance = (baseURL: string) => {
  const instance = axios.create({
    baseURL,
    timeout: config.timeout
  })
  
  // 请求拦截器
  instance.interceptors.request.use(
    (config) => {
      // 添加Token
      const token = localStorage.getItem('access_token')
      if (token) {
        config.headers.Authorization = `Bearer ${token}`
      }
      // 添加客户端类型
      config.headers['X-Client-Type'] = ClientType.ADMIN_WEB
      return config
    },
    (error) => Promise.reject(error)
  )
  
  // 响应拦截器
  instance.interceptors.response.use(
    (response) => {
      const { code, message, data } = response.data
      if (code === 200) {
        return data
      }
      return Promise.reject(new Error(message))
    },
    (error) => {
      // 统一错误处理
      if (error.response?.status === 401) {
        // Token过期，跳转登录
        window.location.href = '/login'
      }
      return Promise.reject(error)
    }
  )
  
  return instance
}
```

```typescript
export function formatDateTime(date: string | Date): string {

  const d = new Date(date)
  return d.toLocaleString('zh-CN', {
    year: 'numeric',
    month: '2-digit',
    day: '2-digit',

// utils/loadAsset.ts
/**
 * 加载资源
 * 优先从OSS加载，失败则使用本地资源
 */
export async function loadAsset(
  ossUrl: string,
  localPath: string
): Promise<string> {
  try {
    const response = await fetch(ossUrl, { method: 'HEAD' })
    if (response.ok) {
      return ossUrl
    }
  } catch (error) {
    console.warn('OSS资源加载失败，使用本地资源:', error)
  }
  return localPath
}
```

```typescript
// 路由前置守卫
router.beforeEach((to, from, next) => {
  // 设置页面标题
  document.title = `${to.meta.title || ''} - 飞机订票系统后台`
  
  // 获取Token
  const token = localStorage.getItem('access_token') || localStorage.getItem('token')
  
  // 白名单
  const whiteList = ['/login', '/oauth2/callback']
  
  if (whiteList.includes(to.path)) {
    // 白名单页面直接放行
    next()
  } else if (!token) {
    // 未登录，跳转登录页
    next('/login')
  } else {
    // 已登录，放行
    next()
  }
})
```

## 9. 总结

ATBS管理端前端架构具有以下特点：

1. **分层清晰**：表现层、业务逻辑层、数据访问层、基础设施层职责分明
2. **组件化**：高内聚低耦合的组件设计，提高复用性
3. **类型安全**：TypeScript全栈类型约束，减少运行时错误
4. **状态管理**：Pinia提供简洁高效的状态管理
5. **工程化**：ESLint + Prettier代码规范，Git版本控制

这种架构设计适合中大型管理后台项目的长期发展和维护。
